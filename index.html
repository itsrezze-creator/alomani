Pastikan Minecraft Bedrock terpasang di perangkat ini. Fitur addExternalServer tidak didukung pada beberapa konsol. Sebagai alternatif, buka Minecraft → Play → Servers → Add Server dan masukkan detail di atas secara manual.

const SERVER_ADDR = "193.176.28.106"; // ganti sesuai kebutuhan const SERVER_PORT = 30037; // ganti sesuai kebutuhan // ================================= // Tampilkan detail di UI document.getElementById('srvName').textContent = SERVER_NAME; document.getElementById('srvAddr').textContent = SERVER_ADDR; document.getElementById('srvPort').textContent = SERVER_PORT; // Bangun URI deep link Bedrock: minecraft://?addExternalServer=Nama|alamat:port const buildUri = (scheme = 'minecraft') => `${scheme}://?addExternalServer=${encodeURIComponent(SERVER_NAME)}|${SERVER_ADDR}:${SERVER_PORT}`; // Coba buka lewat beberapa cara agar kompatibel di berbagai browser/perangkat function openMinecraft() { const uri = buildUri('minecraft'); const fallbackUri = buildUri('minecraftpe'); // skema lama di beberapa perangkat // 1) Coba set location langsung let opened = false; try { window.location.href = uri; opened = true; } catch (e) {} // 2) Coba via iframe tersembunyi sebagai cadangan setTimeout(() => { if (document.visibilityState === 'visible') { const iframe = document.createElement('iframe'); iframe.style.display = 'none'; iframe.src = uri; document.body.appendChild(iframe); } }, 150); // 3) Jika masih gagal (app tidak menangkap skema), coba skema lama setTimeout(() => { if (document.visibilityState === 'visible') { window.location.href = fallbackUri; } }, 700); // Update status UI const status = document.getElementById('status'); status.innerHTML = 'Mencoba membuka Minecraft dan menambahkan server…'; } // Tombol manual document.getElementById('openLink').addEventListener('click', (e) => { e.preventDefault(); openMinecraft(); }); document.getElementById('retry').addEventListener('click', openMinecraft); // Link tombol utama diarahkan ke skema, agar bisa ditekan manual jika perlu document.getElementById('openLink').href = buildUri('minecraft'); // Jalankan otomatis saat halaman dimuat window.addEventListener('load', () => { setTimeout(openMinecraft, 400); });
